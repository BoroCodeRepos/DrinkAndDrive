322 private void CreatePlayerList(out List<Player> list)
323 {
324 list = new List<Player>();
325 XmlNodeList playerNodes = resources.document.GetElementsByTagName("player");
326 // utworzenie wewnętrznej listy z graczami
327 foreach (XmlNode node in playerNodes)
328 {
329 string name = node.Attributes["name"].Value;
330 int score = Convert.ToInt32(node.Attributes["score"].Value);
331 float time = float.Parse(node.Attributes["time"].Value);
332 list.Add(new Player(name, score, time));
333 }
334 // sortowanie listy
335 list.Sort(
336 delegate (Player A, Player B)
337 {
338 if (A.score > B.score) return -1;
339 else if (A.score < B.score) return 1;
340 return 0;
341 }
342 );

394 private void LoadPlayersResults(List<Player> list, int page, int itemsPerPage)
395 {
396 // określenie maksymalnej długości tekstu wyświetlanego w miejscu nazwy gracza
397 int maxNameLen = 10;
398 // dodanie elementów graczy
399 for (int i = 0; i < itemsPerPage; i++)
400 {
401 // wyznaczenie identyfikatora l.p. zależnego od aktualnej strony
402 int id = i + itemsPerPage * (page - 1);
403 // pobranie nazwy i sprawdzenie warunku maksymalnej długości tekstu
404 string name = list[id].name.Trim();
405 if (name.Length > maxNameLen)
406 {
407 // tekst jest zbyt długi
408 string newName = "";
409 // wyznaczenie nowego stringu - kopia pierwszych znaków
410 for (int j = 0; j < maxNameLen - 3; j++)
411 newName += name[j];
412 // oraz dodanie ... na końcu wyświetlanego tekstu
413 name = newName.Trim() + "...";
414 }
415 // dodanie komponentów do listy
416 components.Add(new GUI.Text(new Vector2f(250f, 230f + i * 80f), 40, $"{name}", resources.font, idle));
417 components.Add(new GUI.Text(new Vector2f(500f, 230f + i * 80f), 40, $"{list[id].score}", resources.font, idle));
418 components.Add(new GUI.Text(new Vector2f(830f, 230f + i * 80f), 40, $"{list[id].time}", resources.font, idle));
419 components.Add(new GUI.Text(new Vector2f(60f, 230f + i * 80f), 40, $"{id + 1}.", resources.font, idle));

429 private void Save(Text text, int score, double time)
430 {
431 // walidacja wprowadzonego tekstu
432 if (text.DisplayedString.Trim().Length > 0)
433 {
434 // pobranie globalnego czasu gry wszystkich graczy
435 XmlAttribute totalTimeAttr = resources.document.GetElementsByTagName("game")[0].Attributes["total_time"];
436 int totalTime = Convert.ToInt32(totalTimeAttr.Value);
437 // utworzenie węzła graczy
438 XmlNode playersNode = resources.document.GetElementsByTagName("players")[0];
439 // utworzenie nowego elementu XML
440 XmlElement newPlayer = resources.document.CreateElement("player");
441 // utworzenie atrybutów gracza
442 newPlayer.SetAttribute("name", text.DisplayedString);
443 newPlayer.SetAttribute("score", score.ToString());
444 newPlayer.SetAttribute("time", ((int)time).ToString());
445 // zwiększenie globalnego czasu gry
446 totalTime += (int)time;
447 // zapis globalnego czasu do XML
448 totalTimeAttr.InnerText = totalTime.ToString();
449 // dodanie nowego gracza do XML
450 playersNode.AppendChild(newPlayer);
451 // zapis dokument XML do tego samego pliku
452 resources.document.Save(C.XML_DOCUMENT_PATH);
453 // ponowny start gry
454 engine.OnStartAgain();
455 }
456 }
