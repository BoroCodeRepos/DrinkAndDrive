144 public void UpdateAnimation(float dt)
145 {
146 List<TYPE> types = new List<TYPE>
147 {
148 TYPE.HEART,
149 TYPE.COIN,
150 TYPE.BEER
151 };
152
153 foreach (var type in types)
154 {
155 // przejscie przez wszystkie animowane elementy (taka sama procedura)
156 var animation = this.animation[type];
157 // zwiększenie czasu trawnia ramki
158 animation.currentFrameTime += dt;
159 // sprawdzenie maksymalnego czasu trwania ramki
160 if (animation.currentFrameTime >= animation.maxFrameTime)
161 {
162 // czas dobiegł końca - kolejna ramka
163 animation.currentFrameTime = 0f;
164 animation.currentFrameId += 1;
165 // sprawdzenie warunku końca ramek
166 if (animation.currentFrameId >= animation.framesNr)
167 animation.currentFrameId = 0;
168 }
169 // ustalenie parametrów pod kątem nowo obliczonych wartości animacji
170 animation.currentFrame.Left = animation.currentFrameId * animation.currentFrame.Width;
171 this.animation[type] = animation;
172 // pobranie z listy wszystkich elementów tylko te o danym typie
173 var animatedEntities = entities.Where(entity => entity.type == type);
174 // i zamiana w tych elementach tekstury
175 foreach (var item in animatedEntities)
176 item.sprite.TextureRect = animation.currentFrame;
177 }

363 Random rnd = new Random();

374 DIRECTION dir = (lane < 2) ? DIRECTION.DOWN : DIRECTION.UP;

378 Vector2f position = new Vector2f(lanes[lane].Left + lanes[lane].Width / 2f, 0f);
379 // sprawdzenie procentowych szans na utworzenie elementu i utworzenie go

406 private void CheckPercentageChances(TYPE type, DIRECTION dir, double percent, Vector2f position)
407 {
408 // utworzenie obiektu danego typu jeśli zostanie poprawnie określony procentowo
409 bool create = false;
410 // pobranie szansy procentowej w zależności od typu elementu
411 double chances = 0d;
412 if (type == TYPE.HEART)
413 chances = heartPercent;
414 if (type == TYPE.COIN)
415 chances = coinPercent;
416 if (type == TYPE.BEER)
417 chances = beerPercent;
418 if (type == TYPE.CAR)
419 chances = carPercent;
420 // obsługa dla serc nie jest objęta timerem
421 if (type != TYPE.HEART)
422 {
423 // pobranie timera danego typu
424 var timer = timers[type];
425 // jeżeli timer jest w użytku sprawdzamy czy doszło do przekroczenia czasu zdarzenia
426 if (timer.IsRun())
427 {
428 // sprawdzenie czy doszło do zdarzenia
429 if (timer.GetEventStatus())
430 {
431 // jeżeli doszło to tworzymy element i zerujemy status zdarzenia.
432 create = true;
433 timer.ClearEventStatus();
434 }
435 }
436 else
437 {
438 // timer nie został jeszcze załączony - pierwsze tworzenie elementu
439 create = true;
440 timer.Start();
441 }
442 }
443 else
444 {
445 // w przypadku serca bezpośrednio następuje tworzenie elementu
446 create = true;
447 }
448 // jeżeli jest zezwolenie na tworzenie i zgadza się warunek na szanse to tworzymy element
449 if (create && chances > percent)
450 CreateElement(position, type, dir);
451 }
