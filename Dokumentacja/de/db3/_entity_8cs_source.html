132 public void UpdateMove(float dt, float speed, float offset)
133 {
134 if (movement != null)
135 {
136 // opis modelu zaawansowanego ruchu
137 dt /= 1000f;
138 // pobranie aktualnej prędkości
139 var velocity = movement.Update(dt);
140 // ustawienie rotacji i przesunięcia
141 Move(velocity.X * dt, velocity.Y * dt);
142 SetRotation(velocity.X);
143 }
144 else
145 {
146 // brak obsługi zaawansowanego ruchu
147 // sprawdzenie czy jest używany efekt
148 // efekt skalowania nie występuje dla samochodów
149 if (effect && type != TYPE.CAR)
150 {
151 // gdy efekt występuje - zmiana skali aż do całkowitego zaniku elementu
152 Vector2f scale = sprite.Scale;
153 scale.X -= 0.08f;
154 scale.Y -= 0.08f;
155 // po całkowitym zaniku usuwamy element
156 if (scale.X < 0f || scale.Y < 0f)
157 {
158 toDelete = true;
159 }
160 else
161 {
162 sprite.Scale = scale;
163 damageBox.Scale = scale;
164 }
165 }
166 else if (effect && type == TYPE.CAR)
167 return;
168 // przesunięcie elementu zgodnie z osią X (offset) i Y (speed)
169 Vector2f position = damageBox.Position;
170 speed = ((dir == DIR.UP) ? 0.6f : 1.5f) * Math.Abs(speed) * dt;
171 position.X = primaryPosX + offset;
172 position.Y += speed;
173 SetPosition(position.X, position.Y);
174 }
175 }
