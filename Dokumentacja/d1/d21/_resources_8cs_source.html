136 private void LoadOthers()
137 {
138 // utworzenie obiektu
139 keys = new KeyBindings();
140 // zablokowanie dostępu do xml
141 xmlAccess.WaitOne();
142 XmlNodeList keyBindingsList = document.GetElementsByTagName("keybindings");
143 xmlAccess.ReleaseMutex();
144 // pobranie wartości i odblokowanie xml
145 foreach (XmlNode keyBindings in keyBindingsList)
146 {
147 // zapisanie wszystkich elementów w kolekcji
148 int key = Convert.ToInt32(keyBindings.Attributes["key"].Value);
149 string value = keyBindings.Attributes["value"].Value;
150 keys[(Keyboard.Key)key] = value;
151 }

169 var key = filename.Split('.')[0];
170 // utworzenie obiektu
171 sounds[key] = new Sound

305 private void InitCarsCollection()
306 {
307 try
308 {
309 // utworzenie obiektu
310 carCollection = new Entities();
311 // blokada dostępu do XML
312 xmlAccess.WaitOne();
313 XmlNodeList carsList = document.GetElementsByTagName("car");
314 xmlAccess.ReleaseMutex();
315 // zwolnienie dostępu do XML
316 IntRect textureRect, damageRect;
317 foreach (XmlNode car in carsList)
318 {
319 // przegląd przez wszystkie samochody i zapisanie ich obiektów w pamięci
320 // pobranie i analiza wartości IntRect z atrybutów
321 textureRect = ParseAttributeRect(car.Attributes["texture_rect"]);
322 damageRect = ParseAttributeRect(car.Attributes["damage_rect"]);
323 // utworzenie obiektu pojazdu
324 Entity entity = new Entity(GetTexture(TYPE.CAR), textureRect, damageRect, TYPE.CAR);
325 // utworzenie movement komponentu
326 entity.CreateMovementCompontent(
327 ParseAttributeMoving(car.Attributes["acceleration"]),
328 ParseAttributeMoving(car.Attributes["deceleration"]),
329 ParseAttributeMoving(car.Attributes["max_velocity"])
330 );
331 // dodanie do kolekcji
332 carCollection.Add(entity);
333 }
334 }
335 catch (Exception exception)
336 {
337 Program.ShowError(exception);
338 }

365 private IntRect ParseAttributeRect(XmlAttribute attribute)
366 {
367 // analiza wartości atrybutu XML i przejście na wartości IntRect
368 string[] values = attribute.Value.Split(' ');
369 IntRect rect;
370 // pobranie kolejnych wartości z atrybutu
371 rect.Left = Convert.ToInt16(values[0]);
372 rect.Top = Convert.ToInt16(values[1]);
373 rect.Width = Convert.ToInt16(values[2]);
374 rect.Height = Convert.ToInt16(values[3]);
375 return rect;
376 }
377
383 private Vector2f ParseAttributeMoving(XmlAttribute attribute)
384 {
385 // analiza atrybutu XML i przejście na wartości wektora
386 string[] values = attribute.Value.Split(' ');
387 Vector2f vector;
388 // pierwsza zmienna X, druga Y
389 vector.X = float.Parse(values[0]);
390 vector.Y = float.Parse(values[1]);
391 return vector;
392 }
