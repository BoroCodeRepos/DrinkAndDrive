263 private void UpdateAlcoTime(float dt)
264 {
265 // aktualizacja obecnego poziomu alkoholu
266 // zwiększenie lub zmniejszenie
267 // do wartości zadanej (alcoLevel)
268 if (currLevel > alcoLevel)
269 {
270 currLevel -= alcLevelStep;
271 if (currLevel < alcoLevel)
272 currLevel = alcoLevel;
273 }
274 else if (currLevel < alcoLevel)
275 {
276 currLevel += alcLevelStep;
277 if (currLevel > alcoLevel)
278 currLevel = alcoLevel;
279 }
280 // aktualizacja timerów
281 alcoTime.Update(dt);
282 alcoTimeToStep.Update(dt);
283 // sprawdzenie wystąpienia zdarzenia do zmniejszenia poziomu alkoholu
284 if (alcoTimeToStep.GetEventStatus())
285 {
286 alcoTimeToStep.ClearEventStatus();
287 // zmniejszenie poziomu alkoholu
288 alcoLevel -= 0.1f;
289 // walidacja wartości
290 if (alcoLevel <= 0f)
291 {
292 alcoLevel = 0f;
293 // stop zliczania czasu po alkoholu
294 alcoTime.Stop();
295 // stop i zerowanie czasu odliczania do kroku
296 alcoTimeToStep.Stop();
297 alcoTimeToStep.ClearTime();
298 // ustawienie niewidzialności filtru
299 filter.SetVisible(false);
300 }
301 // obliczenie wartości nowej skali filtru
302 filter.CalcScale(currLevel);
303 }

680 {
681 // pobranie czasu jazdy po alkoholu

683 // wyznaczenie amplitudy drgań
684 float amp = 10 * (currLevel + score / 1000f) % resources.background.Origin.X;
