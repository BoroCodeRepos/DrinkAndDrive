440 public Input(

442 uint characterSize,

444 Color textColor,
445 Dictionary<Keyboard.Key, string> keys,
446 uint maxLen = 21
447 ) : base(characterSize, null, font, textColor, new Color(Color.Transparent))
448 {
449 // przypisanie wartości początkowych
450 this.maxLen = maxLen;
451 keyStates = new Dictionary<Keyboard.Key, bool>();
452 this.keys = keys;
453 foreach (var key in keys.Keys.ToList())
454 keyStates[key] = true;
455 // utworzenie obiektu tekstu
456 text = new SFML.Graphics.Text(null, font)
457 {
458 CharacterSize = characterSize,
459 Position = new Vector2f(position.X, position.Y),
460 FillColor = new Color(textColor),
461 OutlineColor = new Color(Color.Black),
462 OutlineThickness = 1f,
463 };

465
470 public override void Update(ref RenderWindow window)
471 {
472 // sprawdzenie warunku zatwierdzenia (ENTER)
473 if (Keyboard.IsKeyPressed(Keyboard.Key.Return))
474 onEnter?.Invoke();
475 // przelot przez wszystkie wartości kluczy
476 // obsługiwanych przycisków
477 var keyList = keys.Keys.ToList();
478 foreach (var key in keyList)
479 {
480 // pobranie potrzebnych wartości
481 // obecnego stanu, poprzedniego oraz wartości klawisza
482 var value = keys[key];
483 var currKeyState = Keyboard.IsKeyPressed(key);
484 var oldKeyState = keyStates[key];
485 // sprawdzenie warunku naciśnięcia klawisza
486 if (currKeyState && !oldKeyState)
487 {
488 string str = text.DisplayedString;
489 // w przypadku backspace usuwamy ostatni znak
490 if (value == "BackSpace")
491 {
492 text.DisplayedString = str.Remove(str.Length - 1);
493 }
494 else
495 {
496 // sprawdzenie warunku dużych liter
497 if (!Keyboard.IsKeyPressed(Keyboard.Key.LShift) &&
498 !Keyboard.IsKeyPressed(Keyboard.Key.RShift))
499 value = value.ToLower();
500 // jeżeli nie przekroczono maksymalnej długości to zwiększamy tekst
501 if (str.Length < maxLen)
502 text.DisplayedString = str + value;
503 }
504 }
505 // przypisanie poprzedniej wartości
506 keyStates[key] = currKeyState;
507 }
508 }
513 public override void Render(ref RenderWindow window)
514 {
515 // wyświetlenie komponentu w oknie
516 window.Draw(text);
517 }
518 }
